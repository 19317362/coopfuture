243 void Scheduler::loop() noexcept {
244  boost::context::fcontext_t junkContext;
245  boost::context::fcontext_t exitContext;
246 
247  try {
248  while (ready.size() == 0)
249  if (tasks.size() > 0) {
250  auto task = tasks.front();
251  tasks.pop_front();
252  task();
253  } else
254  innerCallback();
255  } catch (FutureAlreadyFulfilled) {
256  std::cerr << "A Future was fulfilled a second time within a "
257  "Scheduler loop. This is non-recoverable.\n";
258  std::terminate();
259  }
260  exitContext = ready.front();
261  ready.pop_front();
262  boost::context::jump_fcontext(&junkContext, exitContext, 0);
263 }

481 ValueType Future<ValueType, ErrorType>::await() throw(ErrorType) {
482  switch (state) {

484  case State::SUCCESS:
485  return contents.result;
